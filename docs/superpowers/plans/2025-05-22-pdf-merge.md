### Task 2: Implement Merge Command

**Files:**
- Create: `src/commands/merge.ts`

- [ ] **Step 1: Create the merge command file**
Implement the command with validation and Ghostscript execution.

```typescript
// src/commands/merge.ts
import { Command } from 'commander'
import { existsSync } from 'node:fs'
import { resolve } from 'node:path'
import chalk from 'chalk'
import ora from 'ora'
import { getGsPath } from '../utils/platform.js'
import { runGhostscript } from '../utils/ghostscript.js'

export function mergeCommand() {
  return new Command('merge')
    .description('Merge multiple PDF files into one')
    .argument('<files...>', 'Input PDF files to merge')
    .requiredOption('-o, --output <path>', 'Output PDF file path')
    .action(async (files: string[], options: { output: string }) => {
      if (files.length < 2) {
        console.error(chalk.red('\nError:'), 'Provide at least two files to merge.')
        process.exit(1)
      }

      // Validate inputs
      for (const file of files) {
        if (!existsSync(file)) {
          console.error(chalk.red('\nError:'), `Input file not found: ${file}`)
          process.exit(1)
        }
        if (!file.toLowerCase().endsWith('.pdf')) {
          console.error(chalk.red('\nError:'), `File is not a PDF: ${file}`)
          process.exit(1)
        }
      }

      const spinner = ora('Merging PDFs...').start()
      try {
        const gsPath = await getGsPath()
        await runGhostscript({
          gsPath,
          input: files.map(f => resolve(f)),
          output: resolve(options.output)
        })
        spinner.succeed(chalk.green(`Successfully merged into ${options.output}`))
      } catch (err) {
        spinner.fail(chalk.red('Merge failed'))
        throw err
      }
    })
}
```
